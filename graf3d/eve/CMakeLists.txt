if(MSVC)
  ROOT_STANDARD_LIBRARY_PACKAGE(Eve
    HEADERS
      ${EVE_HEADERS1}
    NO_INSTALL_HEADERS
    SOURCES
      ${EVE_SOURCES}
    LINKDEF
      LinkDef1.h
    LIBRARIES
      OpenGL::GL
      OpenGL::GLU
      ${FTGL_LIBRARIES}
      RGlew
    DEPENDENCIES
      Core
      EG
      Ged
      Geom
      RGL
      Physics
  )

  ROOT_GENERATE_DICTIONARY(G__Eve2
    ${EVE_HEADERS2}
    LINKDEF
      LinkDef2.h
    MODULE
      Eve
    MULTIDICT
  )

  ROOT_INSTALL_HEADERS()
else()
  if(MACOSX_GLU_DEPRECATED)
    add_definitions(-Wno-deprecated-declarations)
  endif()

  ROOT_STANDARD_LIBRARY_PACKAGE(Eve
    HEADERS
      ${EVE_HEADERS1}
      ${EVE_HEADERS2}
    SOURCES
      ${EVE_SOURCES}
    LIBRARIES
      OpenGL::GL
      OpenGL::GLU
      ${FTGL_LIBRARIES}
      RGlew
    DEPENDENCIES
      Core
      EG
      Ged
      Geom
      RGL
      Physics
  )
endif()

target_include_directories(Eve PRIVATE
  ${FTGL_INCLUDE_DIRS})
