if(MSVC)
  ROOT_LINKER_LIBRARY(Eve
    ${EVE_SOURCES}
    LIBRARIES
      OpenGL::GL
      OpenGL::GLU
      ${FTGL_LIBRARIES}
      ${GLEW_LIBRARIES}
    DEPENDENCIES
      Core
      EG
      Ged
      Geom
      RGL
      Physics
  )

  ROOT_GENERATE_DICTIONARY(G__Eve1
    ${EVE_HEADERS1}
    NOTARGET
    LINKDEF
      LinkDef1.h
    MODULE
      Eve
    EXTRA_DEPENDENCIES
      G__Eve2.cxx
  )

  ROOT_GENERATE_DICTIONARY(G__Eve2
    ${EVE_HEADERS2}
    NOTARGET
    LINKDEF
      LinkDef2.h
    MODULE
      Eve
    MULTIDICT
  )

  ROOT_GENERATE_ROOTMAP(Eve
    LINKDEF
      LinkDef.h
    DEPENDENCIES
      EG
      Ged
      Geom
      GeomPainter
      Graf
      Graf3d
      Gpad
      Gui
      Hist
      MathCore
      Physics
      RGL
      RIO
      Rint
      Tree
      TreePlayer
  )

  ROOT_INSTALL_HEADERS()
else()
  if(MACOSX_GLU_DEPRECATED)
    add_definitions(-Wno-deprecated-declarations)
  endif()

  ROOT_STANDARD_LIBRARY_PACKAGE(Eve
    HEADERS
      ${EVE_HEADERS1}
      ${EVE_HEADERS2}
    SOURCES
      ${EVE_SOURCES}
    LIBRARIES
      OpenGL::GL
      OpenGL::GLU
      ${FTGL_LIBRARIES}
      ${GLEW_LIBRARIES}
    DEPENDENCIES
      Core
      EG
      Ged
      Geom
      RGL
      Physics
  )
endif()

target_include_directories(Eve PRIVATE
  ${FTGL_INCLUDE_DIRS} ${GLEW_INCLUDE_DIRS})
