\begin{activationtable}
  \code{IDENTITY} & $f(x) = x$ \\
  \code{RELU}     & $f(x) = \begin{cases}
    x, & \text{ if } x \geq 0 \\
    0, & \text{otherwise}\end{cases}$ \\
  \code{TANH}     & $f(x) = tanh(x)$ \\
  \code{SIGMOID}  & $f(x) = \frac{1}{1 + \exp(-x)}$ \\
  \code{SOFTSIGN} & $f(x) = \frac{x}{1 + |x|}$ \\
  \code{GAUSS}    & $f(x) = \exp(-x^2)$ \\
  \code{SYMMRELU} & $f(x) = \begin{cases}
    x, & \text{ if } x \geq 0 \\
    -x, & \text{otherwise}\end{cases}$ \\
\end{activationtable}
