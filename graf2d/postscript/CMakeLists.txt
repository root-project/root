  HEADERS
    TImageDump.h
    TPDF.h
    TPostScript.h
    TSVG.h
    TTeXDump.h
  SOURCES
    src/TImageDump.cxx
    src/TPDF.cxx
    src/TPostScript.cxx
    src/TSVG.cxx
    src/TTeXDump.cxx
  DICTIONARY_OPTIONS
    -writeEmptyRootPCM

    Graf

    set_source_files_properties(src/TPostScript.cxx COMPILE_FLAGS "-Wbidi-chars=none")
